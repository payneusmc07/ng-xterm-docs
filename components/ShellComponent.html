import { Component, OnInit } from "@angular/core"
import { FormControl, FormGroup } from "@angular/forms"
import { SettingsService } from "@services/settings.service"
import { CommandRunnerService } from "@shared/services"
import { Settings } from "@utils/constants"

@Component({
	selector: "ng-xterm-shell-settings",
	template: `
		<div class="form-group" [formGroup]="shellSettingsForm">
			<label class="form-label text-white">*Terminal Session</label>
				<select class="form-control form-control-sm text-white border-secondary"
				        formControlName="shell-type"
				        (change)="setShell(shellSettingsForm.get('shell-type').value)"
				        [value]="shells">
				<option *ngFor="let shell of shells"
				       [ngValue]="shell">{{ shell }}</option>
				</select>
		</div>
	`,
	styles: [ `select {width: 150px;}` ]
})

/** Configure the type of shell spawned by node-pty */
export class ShellComponent implements OnInit {

	/** A type context referring to the type of FormGroup. */
	shellSettingsForm: FormGroup

	/** The available terminal sessions on Mac OS */
	shells: string[] = ["zsh", "bash", "ksh", "tcsh"]


	shellType: string

	constructor(
		private readonly settingsService: SettingsService,
		private readonly commandRunnerService: CommandRunnerService
	) {}

	ngOnInit() {
		this.shellType = this.settingsService.getItem(Settings.SHELL_TYPE) as string

 		// configure the form group which will handle getting/setting the terminal session type.
		this.shellSettingsForm = new FormGroup({
			// an html select used to choose the terminal session type
			"shell-type": new FormControl(this.shellType)
		})
	}

	/**
	 * Set the type of shell session used by the terminal.
	 * @param {string} shell the type of shell session used by the terminal
	 * */
	setShell(shell: string) {
		this.settingsService.setItem(Settings.SHELL_TYPE, shell)
	}
}
