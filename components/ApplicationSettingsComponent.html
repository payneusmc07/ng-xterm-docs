export class ApplicationSettingsComponent implements OnInit {

	/** A type context referring to the type FormGroup. */
	applicationSettingsForm: FormGroup

	/** Should NG-Xterm automatically update its self? */
	shouldAutomaticallyUpdate: boolean

	constructor() {}

	ngOnInit() {
		// configure the application settings form
		this.applicationSettingsForm = new FormGroup({
			"auto-update": new FormControl(),
		})
	}

	* @param {boolean} shouldAutoUpdate if new updates should be automatically
	 * downloaded and installed.

	}

}
