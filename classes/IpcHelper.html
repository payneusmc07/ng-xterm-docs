import { app, BrowserWindow, ipcMain } from "electron"
import { Settings, IPCChannelNames } from "@shared/utils"
import { appSettingsStore } from "./store-factory"

export class IpcHelper {
	constructor() {}

	/**
	 * Handle invocation calls from the ipcRenderer based on the channel
	 * the invocation call will be received on.
	 *
	* @param { BrowserWindow } browserWindow the browser window instance to perform a given action on
	* @param {IPCChannelNames} channel the channel to listen for the given ipc event
	*/
	static handleInvocationFromRenderer(browserWindow: BrowserWindow, channel: IPCChannelNames) {
		switch (channel) {
			// maximize the window
			case IPCChannelNames.MAXIMIZE_WINDOW:
				ipcMain.handle(IPCChannelNames.MAXIMIZE_WINDOW, () => {
					browserWindow.isFullScreen()
						? browserWindow!.setFullScreen(false)
						: browserWindow!.setFullScreen(true)
				})
				break

			// minimize the window
			case IPCChannelNames.MINIMIZE_WINDOW:
				ipcMain.handle(IPCChannelNames.MINIMIZE_WINDOW, () => browserWindow.minimize())
				break

			// close the window
			case IPCChannelNames.CLOSE_WINDOW:
				ipcMain.handle(IPCChannelNames.CLOSE_WINDOW, () => app.quit() )
				break

			// set the window background color
			case IPCChannelNames.SET_BACKGROUND_COLOR:
				ipcMain.handle(IPCChannelNames.SET_BACKGROUND_COLOR, (event, args) => {
					if(process.env.NODE_ENV === "development") {
						console.log("background color set to: ", args)
					}
					appSettingsStore.set(Settings.TERM_BG_COLOR, args)
					browserWindow.setBackgroundColor(String(args))
				})
				break

			// set the window transparency
			case IPCChannelNames.SET_BACKGROUND_TRANSPARENCY:
				ipcMain.handle(IPCChannelNames.SET_BACKGROUND_TRANSPARENCY, async (event, args) => {
					if(process.env.NODE_ENV === "development") {
						console.log("transparency set to: ", args)
					}
					appSettingsStore.set(Settings.TERM_BG_TRANSPARENCY, args)
					browserWindow.setOpacity(args)
				})
				break

			// expand the window to accommodate extra terminal panels
			case IPCChannelNames.EXPAND_WINDOW:
				ipcMain.handle(IPCChannelNames.EXPAND_WINDOW, () => browserWindow.setSize(915, 450))
				break

			// restore the window to its original size when a terminal panel is removed
			case IPCChannelNames.SHRINK_WINDOW:
				ipcMain.handle(IPCChannelNames.SHRINK_WINDOW, () => browserWindow.setSize(580, 450))
				break

			// if no message is received on a valid channel, no nothing
			default:
				return
		}
	}

	/** Remove all active ipcMain listeners */
	static removeIPCListeners() {
		ipcMain.removeAllListeners(IPCChannelNames.SET_BACKGROUND_TRANSPARENCY)
		ipcMain.removeAllListeners(IPCChannelNames.SET_BACKGROUND_COLOR)
		ipcMain.removeAllListeners(IPCChannelNames.RESET_WINDOW)
		ipcMain.removeAllListeners(IPCChannelNames.MINIMIZE_WINDOW)
		ipcMain.removeAllListeners(IPCChannelNames.MAXIMIZE_WINDOW)
		ipcMain.removeAllListeners(IPCChannelNames.CLOSE_WINDOW)
		ipcMain.removeAllListeners(IPCChannelNames.SHRINK_WINDOW)
		ipcMain.removeAllListeners(IPCChannelNames.EXPAND_WINDOW)
	}

}
